#1 desactivar y activar el plugin de edusystem y el wp certificate

#2 ajustarlos documentos automatico colocando el identificador, tipo de documento y el requisito del admin

#3 meter el ajuste de sql de base de datos

SELECT sd.document_id, d.type_file, d.id_requisito
FROM wp_student_documents AS sd
INNER JOIN wp_documents AS d 
    ON sd.document_id = d.name;

UPDATE wp_student_documents AS sd
INNER JOIN wp_documents AS d 
    ON sd.document_id = d.name
SET 
    sd.type_file = d.type_file,
    sd.id_requisito = d.id_requisito;

SELECT sd.document_id, dc.type_file, dc.id_requisito
FROM wp_student_documents AS sd
INNER JOIN wp_documents_certificates AS dc 
    ON sd.document_id = dc.document_identificator;

UPDATE wp_student_documents AS sd
INNER JOIN wp_documents_certificates AS dc 
    ON sd.document_id = dc.document_identificator
SET 
    sd.type_file = dc.type_file,
    sd.id_requisito = dc.id_requisito;
